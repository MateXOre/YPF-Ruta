-- estacion.rs

main:
    inicia la estacion
    se conecta a las otras estaciones de la region -> se establece un lider y el ring 

    # hilo 1: escucha conexiones de clientes
    abre un socket para escuchar conexiones entrantes de clientes
    
    loop:
        establece conexion de cliente                                   
        se fija si hay surtidores libres (hay menos del limite)
        lanza una tarea (actix) para manejar la conexion con el cliente

        SURTIDOR 
        tarea maneja la conexion:
            pedir datos                                                 || pedir_datos_de_cobro || Surtidor -> Cliente
            recibe id_tarjeta y monto a cargar                          || devolver_datos_cobro || Cliente -> Surtidor
            envia solicitud de carga a la estacion                      || cobrar_a_cliente || Surtidor -> Estacion
            espera por el resultado                                     || finalizar_venta || Estacion -> Surtidor
            envia resultado al cliente                                  || resultado_carga || Surtidor -> Cliente    
            cierra la conexion
            se elimina la tarea

    # hilo 2: envia solicitudes de pago
    loop:
        espera por solicitudes de carga de los surtidores (channel)     || cobrar_a_cliente || Surtidor -> Estacion
        envia la solicitud al lider de la region                        || solicitud_cobro || Estacion -> Estacion(Lider)
        /// si sos el lider acumulas la solicitud con la del resto de estaciones
        --- si el lider no responde empezar anillo con algoritmo de eleccion ---
        --- si no tenes internet lo guardas en pagos_offline y envias la confirmacion al surtidor ---  || finalizar_venta || Estacion -> Surtidor
    
    # hilo 3: recibe respuestas de pago
    loop:
        espera por respuestas del lider                                 || confirmar_transacciones || Estacion(Lider) -> Estacion 
        notifica al surtidor correspondiente con el resultado           || finalizar_venta || Estacion -> Surtidor

    # hilo 4: comunicacion con otras estaciones
    escuchar conexiones entrantes de otras estaciones
    por cada conexion entrante
    match mensaje
        solicitud_cobro_como_lider                                      || solicitud_cobro || Estacion -> Estacion(Lider)
            recibe pedido de pago
            lo guarda en una cola de pedidos por informar
        aviso de cambio de Lider                                        || election || Estacion <-> Estacion  
        lider caido                                                     || lider_caido || Estacion <-> Estacion 
        informar pedidos                                                || informar_ventas_offline || Estacion <-> Estacion


    # hilo lider: enviar pedidos y respuestas
        cada 30 segundos se conecta con todas las estaciones de la region mediante el anillo
        toma los pedidos por informar                               || informar_pagos || Estacion(Lider) -> Estacion   ?????? informar_pagos ????????
        envia a ypfruta todos los pedidos                           || validar_ventas || Estacion(Lider) -> YPFRUTA
        esperar la respuesta                                        || transacciones_por_estacion || YPFRUTA -> Estacion(Lider)
        enviar las respuestas a las estaciones correspondientes     || confirmar_transacciones || Estacion(Lider) -> Estacion





-- empresa.rs

main.rs
    match
        - VerGastos(vehiculo)           
            envía solicitud         || gastos_empresa || Empresa -> YPFRUTA 
            espera respuesta        || gastos_empresa_respuesta || YPFRUTA -> Empresa    
            filtra y muestra gastos del vehiculo

        - VerGastosGenerales()
            envía solicitud         || gastos_empresa || Empresa -> YPFRUTA 
            espera respuesta        || gastos_empresa_respuesta || YPFRUTA -> Empresa    
            muestra todos los gastos

        - ConfigurarLimite(vehiculo)
            solicita nuevo limite   || configurar_limite || Empresa -> YPFRUTA 
            espera confirmacion     || confirmacion_limite || YPFRUTA -> Empresa

        - ConfigurarLimiteGeneral()
            solicita nuevo limite   || configurar_limite_general || Empresa -> YPFRUTA 
            espera confirmacion     || confirmacion_limite_general || YPFRUTA -> Empresa
